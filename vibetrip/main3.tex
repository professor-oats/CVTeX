% -----------------------------
% LUA: PURE DATA GENERATOR
% -----------------------------
\directlua{

local function noise(i)
  return math.abs(math.sin(i * 12.9898 + 78.233)) % 1
end

local out = {}

for i=1,300 do
  local x = noise(i*1.3) * 12
  local y = noise(i*2.1) * 18
  local r = 0.02 + noise(i*3.7) * 0.08

  out[#out+1] = string.format("%.3f/%.3f/%.3f", x, y, r)
end

tex.sprint("\\def\\terrainpoints{")
tex.sprint(table.concat(out, ","))
tex.sprint("}")
}

}

}
